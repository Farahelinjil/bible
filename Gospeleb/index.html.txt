    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-app.js";
        import { getFirestore, doc, setDoc, collection, onSnapshot, getDocs, query } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-firestore.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-auth.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'gospel-app-prod';

        window.appState = { db, auth, appId, user: null };

        // تسجيل الدخول أولاً قبل أي عملية أخرى
        const initAuth = async () => {
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (e) {
                console.error("Auth failed:", e);
            }
        };

        onAuthStateChanged(auth, (user) => {
            window.appState.user = user;
            if (user) {
                const status = document.getElementById('auth-status');
                if (status) {
                    status.innerText = "متصل بقاعدة البيانات ✓";
                    status.className = "text-xs text-green-600 font-bold";
                }
                // بدء تحميل البيانات فقط بعد نجاح تسجيل الدخول
                if (window.ui) window.ui.loadInitialData();
            }
        });

        initAuth();
    </script>

    <script type="module">
        import { doc, setDoc, onSnapshot, collection, getDocs, query } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-firestore.js";

        const GOSPEL_DATA = {
            matthew: { name: "بشارة متى", max: 28 },
            mark: { name: "بشارة مرقس", max: 16 },
            luke: { name: "بشارة لوقا", max: 24 },
            john: { name: "بشارة يوحنا", max: 21 }
        };

        const CHUNK_SIZE = 250000;
        let activeChapters = new Set();
        let unsubMeta = null;

        window.ui = {
            loadInitialData() {
                if (unsubMeta) unsubMeta(); // تنظيف أي مستمع قديم
                const { db, appId, user } = window.appState;
                if (!user) return;

                // استخدام المسار الصحيح والمؤمن
                const metaRef = collection(db, 'artifacts', appId, 'public', 'data', 'audio_meta');
                unsubMeta = onSnapshot(metaRef, (snap) => {
                    activeChapters = new Set();
                    snap.forEach(d => activeChapters.add(d.id));
                    if (window.selectedGospel) this.selectGospel(window.selectedGospel);
                }, (error) => {
                    console.error("Firestore Listener Error:", error);
                });
            },

            selectGospel(id) {
                window.selectedGospel = id;
                document.getElementById('chapters-container').classList.remove('hidden');
                document.getElementById('current-gospel-name').innerText = GOSPEL_DATA[id].name;
                document.querySelectorAll('.gospel-card').forEach(c => c.classList.remove('active'));
                document.getElementById(`tab-${id}`).classList.add('active');

                const grid = document.getElementById('chapters-grid');
                grid.innerHTML = '';
                for (let i = 1; i <= GOSPEL_DATA[id].max; i++) {
                    const hasAudio = activeChapters.has(`${id}_${i}`);
                    const btn = document.createElement('div');
                    btn.className = `chapter-btn ${hasAudio ? 'has-audio shadow-sm' : 'opacity-25 grayscale pointer-events-none'}`;
                    btn.innerText = i;
                    if (hasAudio) btn.onclick = () => this.playChapter(id, i);
                    grid.appendChild(btn);
                }
            },

            async playChapter(id, ch) {
                const { db, appId, user } = window.appState;
                if (!user) return alert("بانتظار تأمين الاتصال...");

                document.getElementById('now-playing').innerText = `استماع: ${GOSPEL_DATA[id].name} - الفصل ${ch}`;
                document.getElementById('player-bar').classList.remove('translate-y-full');
                
                try {
                    const chunksRef = collection(db, 'artifacts', appId, 'public', 'data', 'audio_chunks', `${id}_${ch}`, 'parts');
                    const snap = await getDocs(chunksRef);
                    const parts = [];
                    snap.forEach(d => parts[parseInt(d.id)] = d.data().data);
                    const audio = document.getElementById('audio-element');
                    audio.src = parts.join('');
                    audio.play();
                } catch (e) {
                    alert("حدث خطأ في جلب الملف. يرجى المحاولة لاحقاً.");
                }
            },

            toggleAdmin() {
                const panel = document.getElementById('admin-panel');
                panel.classList.toggle('hidden');
                panel.classList.toggle('flex');
            },

            doLogin() {
                if (document.getElementById('admin-key').value === "123") {
                    document.getElementById('admin-login-ui').classList.add('hidden');
                    document.getElementById('admin-upload-ui').classList.remove('hidden');
                } else {
                    alert("الرمز غير صحيح");
                }
            },

            async startUpload() {
                const gospel = document.getElementById('up-gospel').value;
                const chapter = document.getElementById('up-chapter').value;
                const fileInput = document.getElementById('up-file');
                const file = fileInput.files[0];
                const { db, appId, user } = window.appState;

                if (!file || !chapter) return alert("اختر الملف ورقم الفصل أولاً.");
                if (!user) return alert("لم يتم الاتصال بقاعدة البيانات بعد.");

                const btn = document.getElementById('main-upload-btn');
                btn.disabled = true;
                document.getElementById('progress-container').classList.remove('hidden');

                const reader = new FileReader();
                reader.readAsDataURL(file);
                reader.onload = async () => {
                    const base64 = reader.result;
                    const totalParts = Math.ceil(base64.length / CHUNK_SIZE);
                    const docId = `${gospel}_${chapter}`;
                    
                    try {
                        await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'audio_meta', docId), {
                            gospel, chapter: parseInt(chapter), uploadedAt: new Date().toISOString()
                        });

                        for (let i = 0; i < totalParts; i++) {
                            const chunk = base64.substring(i * CHUNK_SIZE, (i + 1) * CHUNK_SIZE);
                            await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'audio_chunks', docId, 'parts', i.toString()), { data: chunk });
                            const p = Math.round(((i + 1) / totalParts) * 100);
                            document.getElementById('progress-bar').style.width = p + '%';
                            document.getElementById('progress-label').innerText = `تم رفع ${p}%...`;
                        }
                        alert("تم الرفع بنجاح!");
                        location.reload();
                    } catch (err) {
                        alert("فشل الرفع: " + err.message);
                        btn.disabled = false;
                    }
                };
            },

            hidePlayer() {
                document.getElementById('player-bar').classList.add('translate-y-full');
                document.getElementById('audio-element').pause();
            }
        };
    </script>
